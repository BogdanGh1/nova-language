Functions -> Function Functions
Function -> function ( Id_List ) { Instructions }
Id_List -> id , Id_List | id
Instructions -> DefineInstruction | DefineGlobalInstruction | AttributionInstruction
DefineInstruction -> let IdList ;
DefineGlobalInstruction -> global let IdList ;
AttributionInstruction -> id = Expression ;
Expression -> T E1
E1 -> + T E1 | #
T -> F T1
T1 -> * F T1 | #
F -> ( E ) | id
